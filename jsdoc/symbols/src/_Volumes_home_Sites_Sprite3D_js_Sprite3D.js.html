 30 /**
 31  * Creates an instance of Sprite3D
 32  *
 33  * @constructor
 34  * @author boblemarin
 35  * @this {Sprite3D}
 36  * @param {object} element The DOM element to wrap the Sprite3D object around. When no element is provided, a empty div is created and added to the document.
 37  */
 38 function Sprite3D( element, flat ) {
 39 	// private variables
 40 	var p,rx,ry,rz,i,
 41 		alpha = 1,
 42 		listeners = {};
 43 	
 44 	// create an empty <div> if no element is provided
 45 	if ( element == null )
 46 	{
 47 		element = document.createElement("div");
 48 	}
 49 	
 50 	// prepare for 3D positionning
 51 	//element.style.webkitTransformStyle = "preserve-3d"; // <- must find a solution for this line not to appear in the code ???!!!???
 52 	element.style.margin = "0px";
 53 	element.style.padding = "0px";
 54 	element.style.position = "absolute";
 55 	
 56 	// trigger hardware acceleration even if no property is set
 57 	element.style.webkitTransform = "translateZ(0px)";	
 58 	
 59 	// debug style
 60 	//element.style.border = '1px solid red';
 61 	
 62 	this.domElement = element;
 63 	this.style = element.style;
 64 	this.children = [];
 65 }

398 /**
399  * Applies position and rotation values
400  * @return {Sprite3D} The reference to this Sprite3D object
401  */
402 Sprite3D.prototype.update = function() { 
403 	p = "translate3D(" + (this.x-this.regX) + "px," + (this.y-this.regY) + "px," + (this.z-this.regZ) + "px) ";
404 	rx = "rotateX(" + this.rotationX + "deg) ";
405 	ry = "rotateY(" + this.rotationY + "deg) ";
406 	rz = "rotateZ(" + this.rotationZ + "deg) ";
407 
408 	if ( this.rotateFirst )
409 		//this.style.webkitTransform = rx + ry + rz + p;
410 		this.style.webkitTransform = rz + ry + rx + p;
411 	else
412 		this.style.webkitTransform = p + rx + ry + rz;
413 		
414 	return this;
415 	
416 	/*
417 	future version, more flexible :
418 	
419 	var transformString = "rx ry rz p";
420 	var tt = transformString;
421 	tt = tt.replace( "p", p );
422 	tt = tt.replace( "rx", rx );
423 	tt = tt.replace( "ry", ry );
424 	tt = tt.replace( "rz", rz );
425 	
426 	this.style.webkitTransform = tt;
427 	
428 	*/
429 };

496 /**
497  * Removes a Sprite3D object from this Sprite3D children.
498  * @param {Sprite3D} child The Sprite3D object to remove
499  * @return {Sprite3D} The reference to the removed Sprite3D object. null if the child was not found in this Sprite3D children list
500  */
501 Sprite3D.prototype.removeChild = function( child ) {
502 	var n = this.children.indexOf( child );
503 	if ( n > -1 ) {
504 		return this.removeChildAt( n );
505 	}
506 	return null;
507 };
508 
509 /**
510  * Removes the nth Sprite3D object from this Sprite3D children.
511  * @param {number} n The index of the Sprite3D object to remove
512  * @return {Sprite3D} The reference to the removed Sprite3D object.
513  */
514 Sprite3D.prototype.removeChildAt = function( n ) {
515 	--this.numChildren;
516 	this.domElement.removeChild( this.children[n].domElement );
517 	return this.children.splice( n, 1 )[0];
518 };

522 /**
523  * Finds and return the Sprite3D object associated with the provided DOM element
524  * @param {object} element The DOM element
525  * @return {Sprite3D} The reference to the associated Sprite3D object. Returns null if no relevant Sprite3D object was found
526  */
527 Sprite3D.prototype.findFromDOMElement = function( element ) {
528 	for ( i = 0; i < this.numChildren; i++ ) {

530 	}
531 	return null;
